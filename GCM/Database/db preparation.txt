USE X6SgPM1fb2;

DROP TABLE IF EXISTS mapsMetaDetails;
DROP TABLE IF EXISTS mapsFiles;
DROP TABLE IF EXISTS mapsSites;
DROP TABLE IF EXISTS mapsTours;
DROP TABLE IF EXISTS tourSites;
DROP TABLE IF EXISTS citiesMetaDetails;
DROP TABLE IF EXISTS citiesMaps;

DROP TABLE IF EXISTS sites;

CREATE TABLE users(username varchar(20), password varchar(40), firstName Text, lastName Text, email Text, phoneNumber Text);
CREATE TABLE sites(siteId int,siteName Text, siteDescription Text, site_x_coordinate float, site_y_coordinate float);
CREATE TABLE tours(tourId int, tourDescription Text, tourSuggestedTimeToVisit int);
CREATE TABLE mapsMetaDetails(mapId int, mapWidth float, mapHeight float,mapDescription Text, map_x_offset float, map_y_offset float, mapPrice double);
CREATE TABLE mapsFiles(mapId int, mapFile Blob);
CREATE TABLE mapsSites(mapId int, siteId int);
CREATE TABLE toursSites(tourId int, siteId int);
CREATE TABLE mapsTours(mapId int, tourId int);
CREATE TABLE citiesMetaDetails(cityId int, cityName Text, cityDescription Text);
CREATE TABLE citiesMaps(cityId int, mapId int);
CREATE TABLE citiesSites(cityId int, siteId int);
